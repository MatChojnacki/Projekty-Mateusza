SELECT i.*, L.NAME AS LIST_NAME
FROM ISSUES i
JOIN ISSUESLISTS L ON i.ISSUELIST_ID = L.ID;

SELECT I.*, U.FIRSTNAME AS USER_NAME, U.LASTNAME AS USER_SURNAME
FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(I.USER_ID_ASSIGNEDTO) AS NUM_TASKS
FROM USERS U
JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.ID;
